"""Core Protocol definitions for Zent framework.

All core abstractions are defined as Python Protocols for zero-cost extension.
"""
from typing import Protocol, runtime_checkable, AsyncIterator
from datetime import datetime
from dataclasses import dataclass, field
from enum import Enum


# ==================== 基础类型 ====================


class MessageRole(str, Enum):
    """Message roles in conversation."""
    SYSTEM = "system"
    USER = "user"
    ASSISTANT = "assistant"
    TOOL = "tool"


@dataclass(frozen=True)
class Message:
    """A message in the conversation."""
    role: MessageRole
    content: str
    metadata: dict = field(default_factory=dict)
    timestamp: datetime = field(default_factory=datetime.now)


@dataclass
class ToolCall:
    """A tool call request from LLM."""
    id: str
    name: str
    arguments: dict


@dataclass
class ToolResult:
    """Result of a tool execution."""
    call_id: str
    output: str
    is_error: bool = False
    duration_ms: float = 0.0


@dataclass
class TokenUsage:
    """Token usage statistics."""
    prompt: int
    completion: int
    total: int


@dataclass
class LLMResponse:
    """Response from an LLM."""
    content: str | None
    tool_calls: list[ToolCall]
    usage: TokenUsage
    model: str
    finish_reason: str | None = None

    Implementations must provide async methods for completion and streaming.

            messages: List of messages in the conversation
            tools: Optional list of tools available to the LLM
            temperature: Sampling temperature (0-2)
            max_tokens: Maximum tokens to generate
            **kwargs: Additional provider-specific parameters
        
        Returns:
            LLMResponse with content, tool calls, and usage stats
        """
        ...
    
    async def stream(
        self,
        messages: list[Message],
        **kwargs
    ) -> AsyncIterator[str]:
        """Stream completion from the LLM.
        
        Args:
            messages: List of messages in the conversation
            **kwargs: Additional provider-specific parameters
        
        Yields:
            Chunks of the response as they become available
        """
        ...

    Tools are callable functions that agents can use to interact with the world.
    """
    
    @property
    def name(self) -> str:

"""Session and Step state management."""
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum, auto
from typing import TYPE_CHECKING

from zent.core.protocols import Message, MessageRole, ToolCall, ToolResult

if TYPE_CHECKING:
    from zent.core.result import AgentResult


class StepType(Enum):
    """Types of execution steps."""
    THINK = auto()      # Reasoning/planning step
    TOOL_CALL = auto()  # Tool invocation
    ANSWER = auto()     # Final answer


@dataclass
class Step:
    """A single step in the agent execution."""
    type: StepType
    content: str
    tool_call: ToolCall | None = None
    result: ToolResult | None = None
    timestamp: datetime = field(default_factory=datetime.now)


@dataclass
class Session:
    """Runtime session state.
    
    Maintains conversation history and execution context.
    """
    query: str
    context: dict = field(default_factory=dict)
    system_prompt: str | None = None
    messages: list[Message] = field(default_factory=list)
    steps: list[Step] = field(default_factory=list)
    created_at: datetime = field(default_factory=datetime.now)
    
    def __post_init__(self):
        """Initialize with system prompt and user query."""
        if self.system_prompt:
            self.messages.append(
                Message(role=MessageRole.SYSTEM, content=self.system_prompt)
            )
        self.messages.append(
            Message(role=MessageRole.USER, content=self.query)
        )
    
    def add_step(self, step: Step) -> None:
        """Add a step and update messages."""
        self.steps.append(step)
        
        # Convert step to message
        if step.type == StepType.THINK:
            # Thoughts are not stored in messages (internal reasoning)
            pass
        elif step.type == StepType.TOOL_CALL and step.result:
            # Add tool call and result as assistant + tool messages
            self.messages.append(
                Message(
                    role=MessageRole.ASSISTANT,
                    content="",
                    metadata={"tool_calls": [step.tool_call]}
                )
            )
            self.messages.append(
                Message(
                    role=MessageRole.TOOL,
                    content=step.result.output,
                    metadata={"tool_call_id": step.result.call_id}
                )
            )
        elif step.type == StepType.ANSWER:
            self.messages.append(
                Message(role=MessageRole.ASSISTANT, content=step.content)
            )
    
    def to_messages(self) -> list[Message]:
        """Get all messages for LLM input."""
        return list(self.messages)
```

"""Tool implementations."""
import inspect
import json
from dataclasses import dataclass
from typing import Callable, get_type_hints

from zent.core.protocols import Tool


@dataclass
class FunctionTool:
    """Tool wrapper for Python functions.
    
    Automatically generates JSON Schema from function signature and docstring.
    """
    func: Callable
    name: str | None = None
    description: str | None = None
    _parameters: dict | None = None
    
    def __post_init__(self):
        """Auto-extract metadata from function if not provided."""
        if self.name is None:
            self.name = self.func.__name__
        
        if self.description is None:
            self.description = inspect.getdoc(self.func) or ""
    
    @property
    def parameters(self) -> dict:
        """Generate JSON Schema from function signature."""
        if self._parameters is not None:
            return self._parameters
        
        sig = inspect.signature(self.func)
        type_hints = get_type_hints(self.func)
        
        properties = {}
        required = []
        
        for param_name, param in sig.parameters.items():
            param_type = type_hints.get(param_name, str)
            
            # Map Python types to JSON Schema types
            json_type = "string"
            if param_type == int:
                json_type = "integer"
            elif param_type == float:
                json_type = "number"
            elif param_type == bool:
                json_type = "boolean"
            elif param_type == list or str(param_type) == "list":
                json_type = "array"
            elif param_type == dict or str(param_type) == "dict":
                json_type = "object"
            
            properties[param_name] = {"type": json_type}
            
            # Add to required if no default
            if param.default == inspect.Parameter.empty:
                required.append(param_name)
        
        self._parameters = {
            "type": "object",
            "properties": properties,
            "required": required
        }
        
        return self._parameters
    
    async def run(self, **kwargs) -> str:
        """Execute the wrapped function."""
        # Check if function is async
        if inspect.iscoroutinefunction(self.func):
            result = await self.func(**kwargs)
        else:
            result = self.func(**kwargs)
        
        # Convert result to string
        if isinstance(result, str):
            return result
        return json.dumps(result)
```

"""Memory implementations."""
from dataclasses import dataclass, field

from zent.core.protocols import Memory, Message


@dataclass
class InMemoryMemory:
    """Simple in-memory memory implementation.
    
    Stores messages in a list. Not persistent across restarts.
    """
    _messages: list[Message] = field(default_factory=list)
    
    async def add(self, message: Message) -> None:
        """Add a message to memory."""
        self._messages.append(message)
    
    async def get(
        self, 
        query: str | None = None, 
        limit: int = 10
    ) -> list[Message]:
        """Get recent messages.
        
        Note: query parameter is ignored for simple in-memory storage.
        For semantic search, use a vector memory implementation.
        """
        # Return most recent messages
        return self._messages[-limit:] if self._messages else []
    
    async def clear(self) -> None:
        """Clear all messages."""
        self._messages.clear()
```

"""Agent coordinator - manages workflow execution."""
from dataclasses import dataclass, field
from typing import List, Optional, Callable

from zent.core.protocols import LLM, Tool, Memory, Planner
from zent.core.session import Session, Step, StepType
from zent.core.result import AgentResult, StreamChunk, ToolResult


@dataclass
class AgentConfig:
    """Configuration for Agent."""
    llm: LLM
    tools: List[Tool] = field(default_factory=list)
    memory: Optional[Memory] = None
    planner: Optional[Planner] = None
    max_iterations: int = 10
    system_prompt: Optional[str] = None
    
    # Callbacks
    on_step: Optional[Callable[[Step], None]] = None
    on_tool: Optional[Callable[["ToolCall", ToolResult], None]] = None


class Agent:
    """Agent coordinator - manages workflow execution.
    
    Usage:
        agent = Agent(config)
        result = await agent.run("Query weather in Tokyo")
    """
    
    def __init__(self, config: AgentConfig):
        self.config = config
        self.tools_map = {t.name: t for t in config.tools}
    
    async def run(self, query: str, **context) -> AgentResult:
        """Execute a single task.
        
        Args:
            query: User query
            **context: Additional context data
        
        Returns:
            AgentResult with output and execution details
        """
        session = Session(
            query=query,
            context=context,
            system_prompt=self.config.system_prompt
        )
        
        try:
            # Get LLM response
            messages = session.to_messages()
            response = await self.config.llm.complete(
                messages=messages,
                tools=self.config.tools if self.config.tools else None
            )
            
            # Handle tool calls if any
            if response.tool_calls:
                # Execute first tool call
                tool_call = response.tool_calls[0]
                tool = self.tools_map.get(tool_call.name)
                
                if tool:
                    tool_result = await tool.run(**tool_call.arguments)
                    
                    # Add assistant message with tool call
                    from zent.core.protocols import Message, MessageRole
                    session.messages.append(
                        Message(
                            role=MessageRole.ASSISTANT,
                            content="",
                            metadata={"tool_calls": [tool_call]}
                        )
                    )
                    
                    # Add tool result
                    session.messages.append(
                        Message(
                            role=MessageRole.TOOL,
                            content=tool_result,
                            metadata={"tool_call_id": tool_call.id}
                        )
                    )
                    
                    # Get final response
                    response = await self.config.llm.complete(
                        messages=session.to_messages()
                    )
            
            # Return result
            return AgentResult(
                output=response.content or "",
                session=session,
                success=True
            )
            
        except Exception as e:
            return AgentResult(
                output=str(e),
                session=session,
                success=False,
                error=e
            )
    
    async def stream(self, query: str, **context):
        """Stream execution (placeholder).
        
        Full streaming implementation will be added in Phase 2.
        """
        result = await self.run(query, **context)
        yield StreamChunk(type="answer", content=result.output)
```

"""OpenAI LLM adapter.

Requires: pip install openai
"""
from typing import AsyncIterator

try:
    from openai import AsyncOpenAI
    HAS_OPENAI = True
except ImportError:
    HAS_OPENAI = False

from zent.core.protocols import LLM, Message, MessageRole, LLMResponse, TokenUsage, ToolCall


class OpenAILLM:
    """OpenAI LLM adapter.
    
    Usage:
        llm = OpenAILLM(model="gpt-4", api_key="sk-...")
        response = await llm.complete(messages=[...])
    """
    
    def __init__(
        self,
        model: str = "gpt-4",
        api_key: str | None = None,
        base_url: str | None = None,
        **kwargs
    ):
        if not HAS_OPENAI:
            raise ImportError(
                "OpenAI support requires: pip install openai"
            )
        
        self.model = model
        self.client = AsyncOpenAI(api_key=api_key, base_url=base_url)
        self.default_params = kwargs
    
    def _messages_to_openai(self, messages: list[Message]) -> list[dict]:
        """Convert Zent messages to OpenAI format."""
        return [
            {
                "role": msg.role.value,
                "content": msg.content,
                **({"name": msg.metadata["name"]} if msg.metadata.get("name") else {})
            }
            for msg in messages
        ]
    
    def _tools_to_openai(self, tools: list) -> list[dict]:
        """Convert Zent tools to OpenAI format."""
        return [
            {
                "type": "function",
                "function": {
                    "name": tool.name,
                    "description": tool.description,
                    "parameters": tool.parameters
                }
            }
            for tool in tools
        ]
    
    async def complete(
        self,
        messages: list[Message],
        tools: list | None = None,
        temperature: float = 0.7,
        max_tokens: int | None = None,
        **kwargs
    ) -> LLMResponse:
        """Get completion from OpenAI."""
        params = {
            **self.default_params,
            "model": self.model,
            "messages": self._messages_to_openai(messages),
            "temperature": temperature,
            **({"tools": self._tools_to_openai(tools)} if tools else {}),
            **({"max_tokens": max_tokens} if max_tokens else {}),
            **kwargs
        }
        
        response = await self.client.chat.completions.create(**params)
        
        choice = response.choices[0]
        message = choice.message
        
        # Extract tool calls
        tool_calls = []
        if message.tool_calls:
            for tc in message.tool_calls:
                import json
                tool_calls.append(ToolCall(
                    id=tc.id,
                    name=tc.function.name,
                    arguments=json.loads(tc.function.arguments)
                ))
        
        return LLMResponse(
            content=message.content,
            tool_calls=tool_calls,
            usage=TokenUsage(
                prompt=response.usage.prompt_tokens,
                completion=response.usage.completion_tokens,
                total=response.usage.total_tokens
            ),
            model=response.model,
            finish_reason=choice.finish_reason
        )
    
    async def stream(
        self,
        messages: list[Message],
        **kwargs
    ) -> AsyncIterator[str]:
        """Stream completion from OpenAI."""
        params = {
            **self.default_params,
            "model": self.model,
            "messages": self._messages_to_openai(messages),
            "stream": True,
            **kwargs
        }
        
        stream = await self.client.chat.completions.create(**params)
        
        async for chunk in stream:
            if chunk.choices and chunk.choices[0].delta.content:
                yield chunk.choices[0].delta.content
```

"""Anthropic Claude LLM adapter.

Requires: pip install anthropic
"""
from typing import AsyncIterator

try:
    from anthropic import AsyncAnthropic
    HAS_ANTHROPIC = True
except ImportError:
    HAS_ANTHROPIC = False

from zent.core.protocols import LLM, Message, MessageRole, LLMResponse, TokenUsage, ToolCall


class AnthropicLLM:
    """Anthropic Claude LLM adapter.
    
    Usage:
        llm = AnthropicLLM(model="claude-3-opus-20240229", api_key="sk-ant-...")
        response = await llm.complete(messages=[...])
    """
    
    def __init__(
        self,
        model: str = "claude-3-sonnet-20240229",
        api_key: str | None = None,
        base_url: str | None = None,
        **kwargs
    ):
        if not HAS_ANTHROPIC:
            raise ImportError(
                "Anthropic support requires: pip install anthropic"
            )
        
        self.model = model
        self.client = AsyncAnthropic(api_key=api_key, base_url=base_url)
        self.default_params = kwargs
    
    def _messages_to_anthropic(self, messages: list[Message]) -> tuple[str | None, list[dict]]:
        """Convert Zent messages to Anthropic format.
        
        Returns: (system_message, list_of_messages)
        """
        system = None
        anthropic_messages = []
        
        for msg in messages:
            if msg.role == MessageRole.SYSTEM:
                system = msg.content
            elif msg.role == MessageRole.USER:
                anthropic_messages.append({
                    "role": "user",
                    "content": msg.content
                })
            elif msg.role == MessageRole.ASSISTANT:
                anthropic_messages.append({
                    "role": "assistant",
                    "content": msg.content
                })
            elif msg.role == MessageRole.TOOL:
                # Tool results go back as user messages
                anthropic_messages.append({
                    "role": "user",
                    "content": f"Tool result: {msg.content}"
                })
        
        return system, anthropic_messages
    
    def _tools_to_anthropic(self, tools: list) -> list[dict]:
        """Convert Zent tools to Anthropic format."""
        return [
            {
                "name": tool.name,
                "description": tool.description,
                "input_schema": tool.parameters
            }
            for tool in tools
        ]
    
    async def complete(
        self,
        messages: list[Message],
        tools: list | None = None,
        temperature: float = 0.7,
        max_tokens: int | None = 1024,
        **kwargs
    ) -> LLMResponse:
        """Get completion from Anthropic."""
        system, anthropic_messages = self._messages_to_anthropic(messages)
        
        params = {
            **self.default_params,
            "model": self.model,
            "messages": anthropic_messages,
            "temperature": temperature,
            "max_tokens": max_tokens or 1024,
            **({"tools": self._tools_to_anthropic(tools)} if tools else {}),
            **kwargs
        }
        
        if system:
            params["system"] = system
        
        response = await self.client.messages.create(**params)
        
        # Extract tool calls
        tool_calls = []
        content = ""
        
        for block in response.content:
            if block.type == "text":
                content += block.text
            elif block.type == "tool_use":
                tool_calls.append(ToolCall(
                    id=block.id,
                    name=block.name,
                    arguments=block.input
                ))
        
        return LLMResponse(
            content=content or None,
            tool_calls=tool_calls,
            usage=TokenUsage(
                prompt=response.usage.input_tokens,
                completion=response.usage.output_tokens,
                total=response.usage.input_tokens + response.usage.output_tokens
            ),
            model=response.model,
            finish_reason=response.stop_reason
        )
    
    async def stream(
        self,
        messages: list[Message],
        **kwargs
    ) -> AsyncIterator[str]:
        """Stream completion from Anthropic."""
        system, anthropic_messages = self._messages_to_anthropic(messages)
        
        params = {
            **self.default_params,
            "model": self.model,
            "messages": anthropic_messages,
            "max_tokens": 1024,
            "stream": True,
            **kwargs
        }
        
        if system:
            params["system"] = system
        
        async with self.client.messages.stream(**params) as stream:
            async for text in stream.text_stream:
                yield text
```

"""Application layer decorators."""
from functools import wraps
from typing import Callable, TypeVar, ParamSpec

from zent.capabilities.tool import FunctionTool

P = ParamSpec("P")
T = TypeVar("T")


def tool(name: str | None = None, description: str | None = None):
    """Decorator to convert a function into a Tool.
    
    Usage:
        @tool
        def search(query: str) -> str:
            \"\"\"Search for items.\"\"\"
            return f"Results for {query}"
        
        @tool(name="custom_name")
        def my_func(x: int) -> int:
            return x * 2
    
    Args:
        name: Optional custom tool name (defaults to function name)
        description: Optional custom description (defaults to docstring)
    
    Returns:
        FunctionTool instance
    """
    def decorator(func: Callable[P, T]) -> FunctionTool:
        return FunctionTool(
            func=func,
            name=name,
            description=description
        )
    
    # Support @tool without parentheses
    if callable(name):
        func = name
        name = None
        return decorator(func)
    
    return decorator


def agent(llm=None, tools=None, memory=None, **kwargs):
    """Decorator to create an Agent-backed class.
    
    Note: This is a simplified version. Full implementation in Phase 2.

                return f"Results: {query}"

        class AgentWrapper(cls):
